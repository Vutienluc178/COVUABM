📋 Hướng dẫn kết nối Google Sheets

Bước 1: Tạo Google Spreadsheet mới

Vào sheets.google.com → Tạo bảng tính mới → Copy ID từ URL (phần giữa /d/ và /edit)

Bước 2: Tạo Google Apps Script

Vào script.google.com → Tạo project mới → Dán code bên dưới:

const SHEET_ID = '12NWtJ6ge4oWAj8oY7CoXydXoOSn6pkxy_W_XEo7kezw';

function doGet(e) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName('TournamentData')
      || SpreadsheetApp.openById(SHEET_ID).insertSheet('TournamentData');
    const lastRow = sheet.getLastRow();
    let tournaments = [];
    if (lastRow > 0) {
      const vals = sheet.getRange(1,1,lastRow,1).getValues();
      vals.forEach(r => { try { if(r[0]) tournaments.push(JSON.parse(r[0])); } catch(e){} });
    }
    return ContentService.createTextOutput(JSON.stringify({ok:true,tournaments}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(e) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:e.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SHEET_ID);
    let sheet = ss.getSheetByName('TournamentData') || ss.insertSheet('TournamentData');
    sheet.clearContents();
    if (data.tournaments && data.tournaments.length > 0) {
      const rows = data.tournaments.map(t => [JSON.stringify(t),t.name,t.status,t.players.length,t.currentRound+'/'+t.rounds,t.date||'']);
      sheet.getRange(1,1,rows.length,6).setValues(rows);
    }
    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(e) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:e.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

Bước 3: Deploy Web App

Deploy → New deployment → Web app → Execute as: Me → Who has access: Anyone → Deploy → Copy URL

Bước 4: Dán URL vào ô cấu hình

Quay lại trang này → nhập URL Apps Script vào ô Apps Script URL bên dưới.

⚠️ Lưu ý CORS: Nếu gặp lỗi CORS, bạn cần publish script với tùy chọn "Anyone, even anonymous".
